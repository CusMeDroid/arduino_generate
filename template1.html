Penerapan Arduino

Berikut langkah awal mengatur aplikasi arduino.

  • Preferences
    Buka berkas > preferences > additional board manager url :
    Salin dan tempel url dibawah
    http://arduino.esp8266.com/stable/package_esp8266com_index.json
    lalu tekan ok.
  • Alat
    Buka alat > papan > board manager > cari esp8266 lalu install.
  • Pilih Board Wemos
    Buka alat > papan > ESP8266 Board > pilih LOLIN(WEMOS) D1 R2 & Mini.

void setup() {
  Serial.begin(115200);
  pinMode(reLay, OUTPUT);
  digitalWrite(reLay, HIGH);
  
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, HIGH);
  
  // Connect to WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);
  
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
  
  // Inisialisasi server
  server.on("/", handleRoot);
  server.on("/on", handleOn);
  server.on("/off", handleOff);
  server.begin();
  Serial.println("HTTP server started");
}
